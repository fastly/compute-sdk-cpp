13// A CRTP class that adapts a Rust-style iterator to a C++ range.
14// CppRng is the C++ class that implements the iterator interface,
15// and RustIt is the Rust iterator type that we are wrapping.
16// The C++ class should implement a `next()` method that returns an
17// `std::optional<value_type>`, where `value_type` is the type of the values
18// being iterated over.
19template <class CppRng, class RustIt> class RustIteratorRange {
20public:
21 class iterator {
22 public:
23 using iterator_category = std::input_iterator_tag;
24 // next() should return an optional<value_type>, so we extract the value
25 // type from that
26 using value_type =
27 typename decltype(std::declval<CppRng>().next())::value_type;
28 using difference_type = std::ptrdiff_t;
29 using pointer = value_type *;
30 using reference = value_type &;
31
32 iterator() : range_(nullptr) {}
33 explicit iterator(RustIteratorRange *range) : range_(range) {
34 if (range) {
35 ++(*this); // Initialize the first value
36 }
37 }
38
39 iterator &operator++() {
40 if (!range_) {
41 return *this; // No range to iterate over
42 }
43
44 auto val = static_cast<CppRng *>(range_)->next();
45 if (val) {
46 cache_.emplace(std::move(*val));
47 } else {
48 range_ = nullptr; // Signal end of iteration
49 }
50 return *this;
51 }
52
53 iterator operator++(int) {
54 iterator tmp = *this;
55 ++(*this);
56 return tmp;
57 }
58
59 value_type &operator*() { return *cache_; }
60
61 bool operator==(const iterator &other) const {
62 return range_ == other.range_;
63 }
64
65 bool operator!=(const iterator &other) const { return !(*this == other); }
66
67 private:
68 RustIteratorRange *range_;
69 std::optional<value_type> cache_;
70 };
71
72 RustIteratorRange(rust::Box<RustIt> iter) : iter_(std::move(iter)) {}
73
74 iterator begin() { return iterator(this); }
75
76 iterator end() { return iterator(nullptr); }
77
78protected:
79 rust::Box<RustIt> iter_;
80};
81} // namespace fastly::detail
